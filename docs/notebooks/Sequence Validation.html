def validate_seq(seq): 
    """
    Checks if the DNA sequence is valid. Returns True if sequence is valid or False otherwise 
    """
    if seq.upper():
        total_counts = seq.count("A") + seq.count("T") + seq.count("G") + seq.count("C")
    else:
        seq = seq.upper()
        total_counts = seq.count("A") + seq.count("T") + seq.count("G") + seq.count("C")
        
    if total_counts == len(seq):
        return True 
    else:
        return False
